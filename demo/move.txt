; Тест команд пересылки
a 0                     ; Указываем на 0-ю ячейку памяти
i 0 1                   ; Записываем 0 в 0-ю ячейку памяти
i 0 2                   ; Записываем 1 в 1-ю ячейку памяти
i 0 3                   ; Записываем 2 в 2-ю ячейку памяти
i 0 0                   ; Записываем 0 в 3-ю ячейку памяти
; Пересылка типа память-память
c 1  0 11 %0  %1        ; Копируем 1 во 1-ю ячейку памяти
; Пересылка типа память-регистр
c 1  0 10 %1  %%0       ; Из 1-й ячейки памяти переносим число(1) в 0 регистр
c 1  0 10 %2  %%1       ; Из 2-й ячейки памяти переносим число(3) в 1 регистр
c 17 0 10 %3  %%0       ; Вывод 16-битного числа из 0 регистра
; Пересылка типа регистр-регистр
c 1  0 00 %%1 %%0       ; Из 1-го регистра переносим число в 0-й регистр
c 17 0 10 %3  %%0       ; Вывод 16-битного числа из 0 регистра
; Пересылка типа регистр-память
c 1  0 01 %%1 %0        ; Из 1-го регистра переносим число(3) в 1-ю ячейку памяти
; Так как вывод только из регистров, то вернем его в 0-й регистр
c 1  0 10 %0  %%0       ; Из 0-й ячейки памяти переносим число(3) в 0 регистр
c 17 0 10 %3  %%0       ; Вывод 16-битного числа из 0 регистра
e 4                     ; Запуск процессора с указанной ячейки памяти
