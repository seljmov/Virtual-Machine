; Тест команд пересылки
a 0                     ; Указываем на 0-ю ячейку памяти
i 0 17                  ; Записываем 17 в 0-ю ячейку памяти
i 0 3                   ; Записываем 3 в 1-ю ячейку памяти
i 0 0                   ; Записываем 4 в 2-ю ячейку памяти
c 1  0 11 %0  %1        ; Обмениваем числа в 1-й и 0-й ячейках памяти [17][3] -> [3][17]
c 1  0 10 %0  %%1       ; Из 0-й ячейки памяти переносим число в 1 регистр [1] -> [3]
c 17 0 10 %2  %%1       ; Вывод 16-битного числа
e 3                     ; Запуск процессора с указанной ячейки памяти
